export const Example = () => {
    /**
     * Loading a compressed Gaussian Splat SOGs file
     */

    const { data: splat } = useSplat('/skull.sog');

    if (!splat) return null;

    return <Entity>
        <Entity>
            <Camera fov={30} position={[0, , -6]}/>
            <OrbitControls zoomRange={[3, 6]} />
        </Entity>
        {/* Create the splat entity */}
        <Entity 
            rotation={[0, 210, 180]} 
            position={[0, -.5, 0]}
            >
            <GSplat asset={splat}/>
        </Entity>
    </Entity>
}

<Example />
