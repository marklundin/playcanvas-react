export const Primitives = () => {

    /**
     * The Render component can render both assets and primtive shapes.
     * Useful for testing and debugging.
     */
    
    const primitives = [
        "box",
        "capsule",
        "cone",
        "cylinder",
        "plane",
        "sphere",
    ]

    const [primitive, setPrimitive] = useState('box')
    const material = useMaterial({ diffuse: 'red' })

    const app = useApp();

    console.log(app.root);

    // Create a button for each primitive
    const buttons = primitives.map(type => (
        <button 
            key={type} 
            onClick={() => setPrimitive(type)}
            className="m-2 px-4 py-2 bg-zinc-800 pointer-events-auto cursor-pointer hover:bg-zinc-600 text-white rounded shadow transition-colors duration-200 font-medium"
        >
            {type}
        </button>
    ))

    return <Entity>
        {/* Create some environment lighting */}
        <EnvironmentLighting />        

        {/* Render the background grid */}
        <Grid />

        {/* Add a shadow catcher to catch the shadows from the model */}
        <ShadowCatcher width={5} depth={5} />

        {/* Create a camera entity */}
        <Entity name='camera' position={[4, 3, 4]}>
            <Camera clearColor='#111111' fov={28} />
            <OrbitControls inertiaFactor={0.07} zoomRange={[6, 10]} pitchRange={[-90, -5]}/>
            <AutoRotate />
        </Entity>

        {/* Create the right box entity */}
        <Entity position={[0, 0.5, 0]}>
            <Render type={primitive} material={material} />
        </Entity>

        <div className='pointer-events-none absolute bottom-0 top-22 right-0 p-18'>
            {buttons}
        </div>
        
    </Entity>
}

<Primitives />
